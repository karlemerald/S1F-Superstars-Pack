reserve CorkKill

function CorkKill	
	ResetObjectEntity(object.entityPos, TypeName[Blank Object], 0, object.xpos, object.ypos)
	Rand(checkResult, 32)
	if checkResult > 15
		CreateTempObject(animalType1, TypeName[Blank Object], object.xpos, object.ypos)
	else
		CreateTempObject(animalType2, TypeName[Blank Object], object.xpos, object.ypos)
	end if
	object[tempObjectPos].priority = PRIORITY_UNKNOWN
	CreateTempObject(TypeName[Smoke Puff], 0, object.xpos, object.ypos)
	object[tempObjectPos].drawOrder = 4
	CreateTempObject(TypeName[Object Score], object[0].value19, object.xpos, object.ypos)
	object[tempObjectPos].drawOrder = 4
	PlaySfx(SfxName[Destroy], 0)

	if stage.debugMode == 0
		saveRAM[999]++
	end if		
	switch object[0].value19
	case 0
		player.score += 100
		break
	case 1
		player.score += 200
		break
	case 2
		player.score += 500
		break
	case 3
	case 4
	case 5
	case 6
	case 7
	case 8
	case 9
	case 10
	case 11
	case 12
	case 13
	case 14
		player.score += 1000
		break
	case 15
		player.score += 0x2710
		break
	end switch
	if object[0].value19 < 15
		object[0].value19++
	end if	
	object.type = TypeName[Blank Object]
end function

event ObjectMain
	switch object.propertyValue
	case 0
		object.drawOrder = object[0].value18
		object.xpos += object.xvel
		object.ypos += object.yvel
		object.frame++
		if object.frame >= 3
			object.frame = 0
		end if
		object.animationTimer++
		if object.animationTimer >= 22
			object.type = TypeName[Blank Object]
		end if		
		object.value40 = 0
		object.value38 = 0
		object.value41 = 0
		object.value39 = 0
		break

	case 1
		object.xpos += object.xvel
		object.ypos += object.yvel
		if object.outOfBounds == 1
			object.type = TypeName[Blank Object]
		end if
		
		if object.value0 == 0
			PlaySfx(SfxName[Explosion], 0)	
			CallFunction(PlayerObject_CorkPos)
			object.value0++
		end if		

		temp0 = 4
		temp1 = 11
		if object.direction == FLIP_X
			FlipSign(temp0)
			FlipSign(temp1)
			temp2 = 7
			temp0 -= temp2
			temp1 -= temp2
		end if
		
		object.value40 = temp0
		object.value38 = -4
		object.value41 = temp1
		object.value39 = 3
		
		object.frame = 3
		if object.xvel < 0
			object.direction = FLIP_X
		end if
		break

	case 2
		if object[0].animation == 3		
			temp4 = 8
			temp5 = 6
			temp7 = temp4		
			temp7 *= temp5	
			temp7 -= 3	
			if object.value0 < temp7
				temp0 = 0
				temp1 = object[0].xpos
				temp1 += 0x320000
				temp2 = object[0].ypos
				temp2 -= 0x20000		
				temp6 = 0	
				while temp0 < temp5
					if object.value0 == temp6
						temp3 = temp0
						temp3 += 4
						CreateTempObject(TypeName[Lightning Spark], 3, temp1, temp2)
						object[tempObjectPos].frame = temp3
						object[tempObjectPos].priority = PRIORITY_ACTIVE
						object[tempObjectPos].value0 = temp1
						object[tempObjectPos].value0 -= object[0].xpos
						object[tempObjectPos].value1 = temp2
						object[tempObjectPos].value1 -= object[0].ypos
					end if	
					temp6 += temp4		
					temp1 += 0x90000	
					temp2 += 0x20000										
					temp0++
				loop	
				object.value0++
			end if
		end if
		if object[0].animation != 3		
			object.type = TypeName[Blank Object]
		end if						
		break

	case 3	
		object.drawOrder = 5
		object.xpos = object[0].xpos
		object.ypos = object[0].ypos
		if object[0].direction == FLIP_NONE
			object.xpos += object.value0	
			object.ypos += object.value1	
		else		
			object.xpos -= object.value0	
			object.ypos -= object.value1	
		end if			
		object.value40 = -12
		object.value38 = -12
		object.value41 = 12
		object.value39 = 12		
		if object.animationTimer == 10
			object.animationTimer = 0
			object.direction ^= 1
		else
			object.animationTimer++
		end if
		if object[0].animation != 3		
			object.type = TypeName[Blank Object]
		end if			
		break

	case 10
		switch object.frame
			case 0
				object.alpha = 224
				break
			case 1
				object.alpha = 255
				break
			case 2
				object.alpha = 224
				break
			case 3
				object.alpha = 192
				break
			case 4
				object.alpha = 160
				break
		end switch
		object.frame++
		object.drawOrder = 6
		
		if object.frame >= 5		
			object.type = TypeName[Blank Object]
		end if	
		break
	end switch

end event


event ObjectDraw
	if object.propertyValue == 10
		DrawRect(0, 0, screen.xsize, screen.ysize, color7, color7, color7, object.alpha)
	else
		if object.propertyValue == 3
			LoadSpriteSheet("Players/DragonTrip2.gif")
			DrawSpriteFX(object.frame, FX_FLIP, object.xpos, object.ypos)
			object.xpos += object[0].xvel
			object.ypos += object[0].yvel		
			LoadSpriteSheet("Global/Items3.gif")			
		else
			switch object.propertyValue 
			case 1
				LoadSpriteSheet("Players/Fang3.gif")
				break
			default	
				LoadSpriteSheet("Global/Items3.gif")
				break
			end switch
			DrawSpriteFX(object.frame, FX_FLIP, object.xpos, object.ypos)
			LoadSpriteSheet("Global/Items3.gif")
		end if	
	end if	
end event
